    <script>
        let currentTripId = 'default_trip';
        let trips = {};
        let currentPoints = [];
        
        function loadTrips() {
            const saved = localStorage.getItem('travel_map_trips');
            if (saved) {
                trips = JSON.parse(saved);
            } else {
                trips = {};
            }
            
            // Загружаем список путешествий
            renderTripsList();
            
            // Загружаем последнее активное
            const lastTrip = localStorage.getItem('active_trip_id');
            if (lastTrip && trips[lastTrip]) {
                currentTripId = lastTrip;
                currentPoints = trips[currentTripId].points || [];
            } else if (Object.keys(trips).length > 0) {
                currentTripId = Object.keys(trips)[0];
                currentPoints = trips[currentTripId].points || [];
            } else {
                currentTripId = 'trip_' + Date.now();
                trips[currentTripId] = { name: 'Моё путешествие', points: [] };
                currentPoints = [];
                saveAllTrips();
            }
            
            document.getElementById('travelNameInput').value = trips[currentTripId].name || '';
            renderPoints();
        }
        
        function renderTripsList() {
            const container = document.getElementById('tripsList');
            container.innerHTML = '';
            for (let id in trips) {
                const card = document.createElement('div');
                card.className = 'trip-card';
                card.textContent = trips[id].name;
                card.onclick = () => {
                    currentTripId = id;
                    currentPoints = trips[currentTripId].points || [];
                    document.getElementById('travelNameInput').value = trips[currentTripId].name;
                    renderPoints();
                    localStorage.setItem('active_trip_id', currentTripId);
                };
                container.appendChild(card);
            }
        }
        
        function saveAllTrips() {
            localStorage.setItem('travel_map_trips', JSON.stringify(trips));
            renderTripsList();
        }
        
        function saveCurrentTrip() {
            const newName = document.getElementById('travelNameInput').value;
            if (!trips[currentTripId]) {
                trips[currentTripId] = { points: [] };
            }
            trips[currentTripId].name = newName || 'Без названия';
            trips[currentTripId].points = currentPoints;
            saveAllTrips();
            alert('Сохранено!');
        }
        
        function addNewTrip() {
            const newId = 'trip_' + Date.now();
            trips[newId] = { name: 'Новое путешествие', points: [] };
            saveAllTrips();
            currentTripId = newId;
            currentPoints = [];
            document.getElementById('travelNameInput').value = 'Новое путешествие';
            renderPoints();
            localStorage.setItem('active_trip_id', currentTripId);
        }
        
        function renderPoints() {
            const container = document.getElementById('points-container');
            container.innerHTML = '';
            
            currentPoints.forEach((point, idx) => {
                const dot = document.createElement('div');
                dot.className = 'travel-point';
                dot.style.left = point.x + 'px';
                dot.style.top = point.y + 'px';
                dot.onclick = (e) => {
                    e.stopPropagation();
                    if (confirm('Удалить точку?')) {
                        currentPoints.splice(idx, 1);
                        renderPoints();
                        saveCurrentTrip();
                    }
                };
                container.appendChild(dot);
                
                const label = document.createElement('div');
                label.className = 'point-label';
                label.style.left = point.x + 'px';
                label.style.top = (point.y - 15) + 'px';
                label.textContent = point.name || 'Точка';
                container.appendChild(label);
                
                // Рисуем пунктирную линию к предыдущей точке
                if (idx > 0) {
                    const prev = currentPoints[idx-1];
                    const line = document.createElement('div');
                    line.className = 'dashed-line';
                    const dx = point.x - prev.x;
                    const dy = point.y - prev.y;
                    const length = Math.sqrt(dx*dx + dy*dy);
                    const angle = Math.atan2(dy, dx) * 180 / Math.PI;
                    line.style.width = length + 'px';
                    line.style.left = prev.x + 'px';
                    line.style.top = prev.y + 'px';
                    line.style.transform = `rotate(${angle}deg)`;
                    container.appendChild(line);
                }
            });
        }
        
        document.getElementById('fantasy-map').addEventListener('click', (e) => {
            const rect = document.getElementById('fantasy-map').getBoundingClientRect();
            let x = e.clientX - rect.left;
            let y = e.clientY - rect.top;
            
            const name = prompt('Название точки:', `Точка ${currentPoints.length + 1}`);
            if (name) {
                currentPoints.push({ x, y, name });
                renderPoints();
                saveCurrentTrip();
            }
        });
        
        function clearAllPoints() {
            if (confirm('Удалить все точки на этой карте?')) {
                currentPoints = [];
                renderPoints();
                saveCurrentTrip();
            }
        }
        
        loadTrips();
    </script>
