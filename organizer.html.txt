    <script>
        // ========== БИЛЕТЫ ==========
        let ticketImages = [];
        let ticketsChecked = false;
        
        function loadTicketsData() {
            const savedImages = localStorage.getItem('ticket_images');
            if (savedImages) ticketImages = JSON.parse(savedImages);
            const savedChecked = localStorage.getItem('tickets_checked');
            ticketsChecked = savedChecked === 'true';
            renderTicketPhotos();
            renderTicketsCheck();
        }
        
        function renderTicketPhotos() {
            const container = document.getElementById('ticketPhotos');
            container.innerHTML = '';
            ticketImages.forEach((imgSrc, idx) => {
                const img = document.createElement('img');
                img.src = imgSrc;
                img.className = 'ticket-img';
                img.onclick = () => {
                    if (confirm('Удалить фото билета?')) {
                        ticketImages.splice(idx, 1);
                        saveTicketImages();
                        renderTicketPhotos();
                    }
                };
                container.appendChild(img);
            });
            const addBtn = document.createElement('div');
            addBtn.className = 'add-ticket-btn';
            addBtn.textContent = '+';
            addBtn.onclick = () => {
                const input = document.createElement('input');
                input.type = 'file';
                input.accept = 'image/*';
                input.onchange = (e) => {
                    const file = e.target.files[0];
                    if (file) {
                        const reader = new FileReader();
                        reader.onload = (event) => {
                            ticketImages.push(event.target.result);
                            saveTicketImages();
                            renderTicketPhotos();
                        };
                        reader.readAsDataURL(file);
                    }
                };
                input.click();
            };
            container.appendChild(addBtn);
        }
        
        function saveTicketImages() {
            localStorage.setItem('ticket_images', JSON.stringify(ticketImages));
        }
        
        function renderTicketsCheck() {
            const circle = document.getElementById('ticketsCheckCircle');
            if (ticketsChecked) {
                circle.classList.add('checked');
            } else {
                circle.classList.remove('checked');
            }
            circle.onclick = () => {
                ticketsChecked = !ticketsChecked;
                localStorage.setItem('tickets_checked', ticketsChecked);
                renderTicketsCheck();
            };
        }
        
        // ========== ОТЕЛЬ ==========
        function loadHotelData() {
            document.getElementById('hotelName').value = localStorage.getItem('hotel_name') || '';
            document.getElementById('hotelAddress').value = localStorage.getItem('hotel_address') || '';
            document.getElementById('hotelDates').value = localStorage.getItem('hotel_dates') || '';
            document.getElementById('hotelNotes').value = localStorage.getItem('hotel_notes') || '';
            
            document.getElementById('hotelName').onchange = () => localStorage.setItem('hotel_name', document.getElementById('hotelName').value);
            document.getElementById('hotelAddress').onchange = () => localStorage.setItem('hotel_address', document.getElementById('hotelAddress').value);
            document.getElementById('hotelDates').onchange = () => localStorage.setItem('hotel_dates', document.getElementById('hotelDates').value);
            document.getElementById('hotelNotes').onchange = () => localStorage.setItem('hotel_notes', document.getElementById('hotelNotes').value);
        }
        
        // ========== СПИСОК ВЕЩЕЙ ==========
        let checklistItems = [];
        
        function loadChecklist() {
            const saved = localStorage.getItem('packing_list');
            if (saved) {
                checklistItems = JSON.parse(saved);
            } else {
                checklistItems = [
                    { text: 'Паспорт', checked: false },
                    { text: 'Билеты', checked: false },
                    { text: 'Зарядное устройство', checked: false }
                ];
            }
            renderChecklist();
        }
        
        function renderChecklist() {
            const container = document.getElementById('checklist');
            container.innerHTML = '';
            checklistItems.forEach((item, idx) => {
                const div = document.createElement('div');
                div.className = 'checklist-item';
                
                const circle = document.createElement('div');
                circle.className = 'checklist-circle' + (item.checked ? ' checked' : '');
                circle.onclick = () => {
                    checklistItems[idx].checked = !checklistItems[idx].checked;
                    saveChecklist();
                    renderChecklist();
                };
                
                const input = document.createElement('input');
                input.type = 'text';
                input.value = item.text;
                input.className = 'checklist-text';
                input.onchange = (e) => {
                    checklistItems[idx].text = e.target.value;
                    saveChecklist();
                };
                
                const del = document.createElement('span');
                del.textContent = '✖';
                del.className = 'delete-item';
                del.onclick = () => {
                    checklistItems.splice(idx, 1);
                    saveChecklist();
                    renderChecklist();
                };
                
                div.appendChild(circle);
                div.appendChild(input);
                div.appendChild(del);
                container.appendChild(div);
            });
        }
        
        function addChecklistItem() {
            checklistItems.push({ text: 'Новая вещь', checked: false });
            saveChecklist();
            renderChecklist();
        }
        
        function saveChecklist() {
            localStorage.setItem('packing_list', JSON.stringify(checklistItems));
        }
        
        // ========== ФИНАНСЫ ==========
        let financeRows = [];
        
        function loadFinance() {
            const saved = localStorage.getItem('finance_rows');
            if (saved) {
                financeRows = JSON.parse(saved);
            } else {
                financeRows = [
                    { category: 'Билеты', amount: 0 },
                    { category: 'Отель', amount: 0 }
                ];
            }
            renderFinance();
        }
        
        function renderFinance() {
            const tbody = document.getElementById('financeBody');
            tbody.innerHTML = '';
            let total = 0;
            financeRows.forEach((row, idx) => {
                const tr = document.createElement('tr');
                const tdCat = document.createElement('td');
                const catInput = document.createElement('input');
                catInput.type = 'text';
                catInput.value = row.category;
                catInput.onchange = (e) => {
                    financeRows[idx].category = e.target.value;
                    saveFinance();
                };
                tdCat.appendChild(catInput);
                
                const tdAmt = document.createElement('td');
                const amtInput = document.createElement('input');
                amtInput.type = 'number';
                amtInput.value = row.amount;
                amtInput.onchange = (e) => {
                    financeRows[idx].amount = parseFloat(e.target.value) || 0;
                    saveFinance();
                    renderFinance();
                };
                tdAmt.appendChild(amtInput);
                
                const tdDel = document.createElement('td');
                const delSpan = document.createElement('span');
                delSpan.textContent = '✖';
                delSpan.className = 'delete-item';
                delSpan.onclick = () => {
                    financeRows.splice(idx, 1);
                    saveFinance();
                    renderFinance();
                };
                tdDel.appendChild(delSpan);
                
                tr.appendChild(tdCat);
                tr.appendChild(tdAmt);
                tr.appendChild(tdDel);
                tbody.appendChild(tr);
                total += row.amount;
            });
            document.getElementById('totalSum').textContent = total + ' ₽';
        }
        
        function addFinanceRow() {
            financeRows.push({ category: 'Новый расход', amount: 0 });
            saveFinance();
            renderFinance();
        }
        
        function saveFinance() {
            localStorage.setItem('finance_rows', JSON.stringify(financeRows));
        }
        
        function deleteFinanceRow(element) {
            // Удаление строки через кнопку
            const row = element.closest('tr');
            const index = Array.from(row.parentNode.children).indexOf(row);
            if (index < financeRows.length) {
                financeRows.splice(index, 1);
                saveFinance();
                renderFinance();
            }
        }
        
        // ========== ОТКРЫТИЕ/ЗАКРЫТИЕ ПАПОК ==========
        document.querySelectorAll('.folder').forEach(folder => {
            const header = folder.querySelector('.folder-header');
            header.addEventListener('click', (e) => {
                e.stopPropagation();
                folder.classList.toggle('open');
            });
        });
        
        // Загрузка всех данных
        loadTicketsData();
        loadHotelData();
        loadChecklist();
        loadFinance();
    </script>
