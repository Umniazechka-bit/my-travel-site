    <script>
        let currentCity = '';
        let notes = [];
        
        function loadCityData() {
            currentCity = localStorage.getItem('current_city') || 'Город';
            document.getElementById('cityName').textContent = currentCity;
            
            // Загружаем заметки для города
            const saved = localStorage.getItem(`city_notes_${currentCity}`);
            if (saved) {
                notes = JSON.parse(saved);
            } else {
                notes = [{
                    title: 'Мой первый день',
                    text: 'Расскажите о вашем путешествии...',
                    images: []
                }];
            }
            renderNotes();
            
            // Загружаем фото
            const savedPhoto = localStorage.getItem(`city_photo_${currentCity}`);
            if (savedPhoto) {
                document.getElementById('cityPhoto').src = savedPhoto;
            }
        }
        
        function renderNotes() {
            const container = document.getElementById('notesContainer');
            container.innerHTML = '';
            
            notes.forEach((note, index) => {
                const noteDiv = document.createElement('div');
                noteDiv.className = 'note-block';
                
                // Заголовок с плюсиком
                const titleDiv = document.createElement('div');
                titleDiv.className = 'note-title';
                const titleInput = document.createElement('input');
                titleInput.value = note.title;
                titleInput.placeholder = 'Название';
                titleInput.onchange = (e) => {
                    notes[index].title = e.target.value;
                    saveNotes();
                };
                const smallPlus = document.createElement('div');
                smallPlus.className = 'small-plus';
                smallPlus.textContent = '+';
                smallPlus.onclick = () => addNewNote();
                titleDiv.appendChild(titleInput);
                titleDiv.appendChild(smallPlus);
                
                // Контент
                const contentDiv = document.createElement('div');
                contentDiv.className = 'note-content';
                
                const textarea = document.createElement('textarea');
                textarea.className = 'note-text';
                textarea.value = note.text;
                textarea.onchange = (e) => {
                    notes[index].text = e.target.value;
                    saveNotes();
                };
                
                const imagesDiv = document.createElement('div');
                imagesDiv.className = 'images-grid';
                
                note.images.forEach((imgSrc, imgIdx) => {
                    const img = document.createElement('img');
                    img.src = imgSrc;
                    img.className = 'city-image';
                    img.onclick = () => {
                        if (confirm('Удалить фото?')) {
                            notes[index].images.splice(imgIdx, 1);
                            saveNotes();
                            renderNotes();
                        }
                    };
                    imagesDiv.appendChild(img);
                });
                
                const addImgBtn = document.createElement('div');
                addImgBtn.className = 'add-image-btn';
                addImgBtn.textContent = '+';
                addImgBtn.onclick = () => {
                    const input = document.createElement('input');
                    input.type = 'file';
                    input.accept = 'image/*';
                    input.onchange = (e) => {
                        const file = e.target.files[0];
                        if (file) {
                            const reader = new FileReader();
                            reader.onload = (event) => {
                                notes[index].images.push(event.target.result);
                                saveNotes();
                                renderNotes();
                            };
                            reader.readAsDataURL(file);
                        }
                    };
                    input.click();
                };
                imagesDiv.appendChild(addImgBtn);
                
                contentDiv.appendChild(textarea);
                contentDiv.appendChild(imagesDiv);
                noteDiv.appendChild(titleDiv);
                noteDiv.appendChild(contentDiv);
                container.appendChild(noteDiv);
            });
        }
        
        function addNewNote() {
            notes.push({
                title: 'Новая заметка',
                text: '',
                images: []
            });
            saveNotes();
            renderNotes();
        }
        
        function saveNotes() {
            localStorage.setItem(`city_notes_${currentCity}`, JSON.stringify(notes));
        }
        
        function changePhoto() {
            const input = document.createElement('input');
            input.type = 'file';
            input.accept = 'image/*';
            input.onchange = (e) => {
                const file = e.target.files[0];
                if (file) {
                    const reader = new FileReader();
                    reader.onload = (event) => {
                        const photoUrl = event.target.result;
                        document.getElementById('cityPhoto').src = photoUrl;
                        localStorage.setItem(`city_photo_${currentCity}`, photoUrl);
                    };
                    reader.readAsDataURL(file);
                }
            };
            input.click();
        }
        
        function goBack() {
            window.location.href = 'index.html';
        }
        
        // Меню для трёх точек
        document.getElementById('photoDots').addEventListener('click', () => {
            const menu = document.getElementById('photoMenu');
            menu.style.display = menu.style.display === 'block' ? 'none' : 'block';
        });
        
        document.addEventListener('click', (e) => {
            if (!e.target.closest('.photo-area')) {
                document.getElementById('photoMenu').style.display = 'none';
            }
        });
        
        loadCityData();
